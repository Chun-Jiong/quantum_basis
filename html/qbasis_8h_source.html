987 // ----------------------------------------------------------------------------
988 
989  // m step of Lanczos (for iram: m = k + np; for simple Lanczos, m depends on the convergence speed)
990  // on entry, assuming k steps of Lanczos already performed
991  // v of length m+1, hessenberg matrix of size m*m (m-step Lanczos)
992  // after decomposition, mat * v[0:m-1] = v[0:m-1] * hessenberg + b[m] * v[m] * e_m^T,
993  // where e_m has only one nonzero element: e[0:m-2] == 0, e[m-1] = 1
994  //
995  // maxit: maximum allowed Lanczos steps, m < maxit
996  // maxit also serves as the leading dimension of hessenberg matrix
997  // a[j] = hessenberg[j+maxit], diagonal of hessenberg matrix
998  // b[j] = hessenberg[j]
999  // a[0] b[1] -> note: b[0] not used
1000  // b[1] a[1] b[2]
1001  // b[2] a[2] b[3]
1002  // b[3] a[3] b[4]
1003  // .. .. .. b[k-1]
1004  // b[k-1] a[k-1]
1005  // on entry:
1006  // a[0], ..., a[k-1] in hessenberg matrix
1007  // b[0], ..., b[k-1], b[k] in hessenberg matrix
1008  // on exit:
1009  // a[0], ..., a[m-1] in hessenberg matrix
1010  // b[0], ..., b[m-1], b[m] in hessenberg matrix
1011  //
1012  // if purpose == "iram":
1013  // on entry:
1014  // v[0], ..., v[k] stored in v
1015  // on exit:
1016  // v[0], ..., v[m] stored in v
1017  //
